clustTest = function(toClust,print=TRUE,scale=TRUE,maxClusts=15,seed=12345,nstart=20,iter.max=100){
  if(scale){ toClust = scale(toClust);}
  set.seed(seed);   
  wss <- (nrow(toClust)-1)*sum(apply(toClust,2,var))
  for (i in 2:maxClusts) wss[i] <- sum(kmeans(toClust,centers=i,nstart=nstart,iter.max=iter.max)$withinss)
  gpw = fviz_nbclust(toClust,
                     kmeans,method="wss",iter.max=iter.max,nstart=nstart,k.max=maxClusts) #alternative way to get wss elbow chart.
  pm1 = pamk(toClust,scaling=TRUE)
  gps = fviz_nbclust(toClust,kmeans,method="silhouette",iter.max=iter.max,nstart=nstart,k.max=maxClusts) 
  if(print){
    grid.arrange(gpw,gps, nrow = 1)
  }
  list(wss=wss,pm1=pm1$nc,gpw=gpw,gps=gps)
}

plotClust = function(km,toClust,discPlot=FALSE){
  nc = length(km$size)
  if(discPlot){par(mfrow=c(2,2))}
  else {par(mfrow=c(3,1))}
  percsize = paste(1:nc," = ",format(km$size/sum(km$size)*100,digits=2),"%",sep="")
  pie(km$size,labels=percsize,col=1:nc)
  
  clusplot(toClust, km$cluster, color=TRUE, shade=TRUE,
           labels=2, lines=0,col.clus=1:nc); #plot clusters against principal components
  
  if(discPlot){
    plotcluster(toClust, km$cluster,col=km$cluster); #plot against discriminant functions ()
  }
  rng = range(km$centers)
  dist = rng[2]-rng[1]
  locs = km$centers+.05*dist*ifelse(km$centers>0,1,-1)
  bm = barplot(km$centers,beside=TRUE,col=1:nc,main="Cluster Means",ylim=rng+dist*c(-.1,.1))
  text(bm,locs,formatC(km$centers,format="f",digits=1))
}

#calculate the market share of a specific combination of profile selection (scenario)
marketshare <- function(set){
  targetdata <- indiv_rating[,set]
  for (i in 1:nrow(targetdata)) {
    targetdata[i,] <- as.numeric(targetdata[i,] == max(targetdata[i,]))
  }
  shares <- c()
  for (i in 1:ncol(targetdata)) {
    shares[i] <- sum(targetdata[,i])/nrow(targetdata)
  }
  MS <- data.frame()
  MS <- rbind(MS, shares)
  colnames(MS) <- set
  return(MS)
}

#write a function to calculate the profit
#share is the data frame produced by function marketshare
#islongrun is a logic variable with T means measure long run revenue, F means short run revenue
#rivalprofile is the rival's prodiuct
profit <- function(shares, islongrun, rivalprofile, ourprofile){
  profit <- 0
  for (i in 1:ncol(shares)) {
    if(colnames(shares)[i] == rivalprofile) next()
    profit <- profit + shares[1,i] * profilesData$margin[which(profilesData$profile == colnames(shares)[i])]*4000
  }
  if(islongrun){
    profit <- profit - (ncol(shares)-1)*20000
  }else{
    profit <- profit - (ncol(shares)-1)*20000
    for (i in colnames(shares)) {
      if(!(i %in% ourprofile) & i != rivalprofile) profit <- profit - 20000/3
    }
  }
  return(profit)
}

####################################################################################
#write a function to calculate the profitability
#share is the data frame produced by function marketshare
#islongrun is a logic variable with T means measure long run revenue, F means short run revenue
#rivalprofile is the rival's prodiuct
#return a data.frame contain the original shares table and a new row with profitability information
profitability <- function(shares, islongrun, rivalprofile, ourprofile){
  profitability <- c()
  for (i in 1:ncol(shares)) {
    unitmargin <- profilesData$margin[as.numeric(colnames(shares)[i])]
    revenue <- unitmargin * shares[1,i] * 4000
    if(islongrun){
      cost <- 20000
    }else{
      if(colnames(shares)[i] %in% ourprofile){
        cost <- 20000 
      }else{
        cost <- 20000 + 20000/3
      }
    }
    profitability[i] <- (revenue - cost)/revenue
  }
  profitability <- rbind(shares, profitability)
  return(profitability)
}

####################################################################################
#write a function to calculate the profitability of the whole scenario
#share is the data frame produced by function marketshare
#islongrun is a logic variable with T means measure long run revenue, F means short run revenue
#rivalprofile is the rival's prodiuct
#return the profitability of the whole scenario
overallprofitability <- function(shares, islongrun, rivalprofile, ourprofile){
  revenue <- 0
  profit <- 0
  for (i in 1:ncol(shares)) {
    if(colnames(shares)[i] == rivalprofile) next()
    profit <- profit + shares[1,i] * profilesData$margin[which(profilesData$profile == colnames(shares)[i])]*4000
  }
  revenue <- profit
  if(islongrun){
    profit <- profit - (ncol(shares)-1)*20000
  }else{
    profit <- profit - (ncol(shares)-1)*20000
    for (i in colnames(shares)) {
      if(!(i %in% ourprofile) & i != rivalprofile) profit <- profit - 20000/3
    }
  }
  return(profit/revenue)
}

allprofit <- function(number_of_profiles, islongrun, rivalprofile, ourprofile){
  conditions <- as.data.frame(combn(16,number_of_profiles))
  profits <- c()
  for (i in 1:ncol(conditions)) {
    if(rivalprofile %in% conditions[[i]]) next()
    set <- conditions[[i]]
    set[number_of_profiles+1] <- rivalprofile
    profits[i] <- profit(marketshare(set), islongrun, rivalprofile, ourprofile)
  }
  profits <- rbind(conditions, profits)
  return(profits)
}
